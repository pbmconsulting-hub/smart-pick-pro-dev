import hashlib, secrets

def generate_tournament_seed():
    """Generate AFTER lock. Cryptographically unpredictable before lock."""
    raw = secrets.token_hex(32)  # 256-bit random
    seed_int = int(hashlib.sha256(raw.encode()).hexdigest()[:8], 16)
    return raw, seed_int

5.2 Step 2: Simulate Tournament Environment (Engine A)
Uses _simulate_single_game() from engine/game_prediction.py with league-average inputs to produce a neutral "tonight's game conditions":
Python
from engine.game_prediction import (
    _simulate_single_game,
    _calculate_expected_possessions,
    _score_from_possession_model,
    LEAGUE_AVG_ORTG,    # 113.0
    LEAGUE_AVG_DRTG,    # 113.0
    LEAGUE_AVG_PACE,    # 100.0
    BLOWOUT_MARGIN_THRESHOLD,  # 15
)

def simulate_tournament_environment(seed_int: int) -> dict:
    """
    Generate one global game environment for the entire tournament.
    
    Uses Engine A (_simulate_single_game) with league-average inputs.
    The output drives Engine B's contextual parameters.
    """
    import random
    random.seed(seed_int)
    
    # League-average matchup with slight variance
    home_ortg = random.gauss(LEAGUE_AVG_ORTG, 3.0)  # 113 ± 3
    away_ortg = random.gauss(LEAGUE_AVG_ORTG, 3.0)
    home_drtg = random.gauss(LEAGUE_AVG_DRTG, 3.0)
    away_drtg = random.gauss(LEAGUE_AVG_DRTG, 3.0)
    home_pace = random.gauss(LEAGUE_AVG_PACE, 2.0)   # 100 ± 2
    away_pace = random.gauss(LEAGUE_AVG_PACE, 2.0)
    
    # Use YOUR existing functions
    expected_poss = _calculate_expected_possessions(home_pace, away_pace)
    
    # Defensive adjustment (from your _score_from_possession_model pattern)
    home_adj_ortg = home_ortg * (LEAGUE_AVG_DRTG / max(away_drtg, 95.0))
    away_adj_ortg = away_ortg * (LEAGUE_AVG_DRTG / max(home_drtg, 95.0))
    
    home_base = _score_from_possession_model(home_adj_ortg, expected_poss)
    away_base = _score_from_possession_model(away_adj_ortg, expected_poss)
    
    # Simulate one full game using YOUR existing quarter-by-quarter engine
    home_score, away_score, went_to_ot = _simulate_single_game(
        home_adj_ortg=home_adj_ortg, home_drtg=home_drtg, home_pace=home_pace,
        away_adj_ortg=away_adj_ortg, away_drtg=away_drtg, away_pace=away_pace,
        expected_possessions=expected_poss,
        home_base_score=home_base, away_base_score=away_base,
    )
    
    # Derive tournament context from the simulated game
    margin = abs(home_score - away_score)
    total = home_score + away_score
    
    # Map to QME input parameters
    if margin > 20:
        blowout_risk = 0.65
        pace_adj = 0.95
    elif margin > BLOWOUT_MARGIN_THRESHOLD:
        blowout_risk = 0.40
        pace_adj = 0.97
    elif margin > 8:
        blowout_risk = 0.20
        pace_adj = 1.0
    elif went_to_ot:
        blowout_risk = 0.05
        pace_adj = 1.08
    else:
        blowout_risk = 0.10
        pace_adj = 1.0
    
    # High/low scoring adjusts pace further
    if total > 235:
        pace_adj *= 1.04
    elif total < 200:
        pace_adj *= 0.96
    
    return {
        "home_score": round(home_score),
        "away_score": round(away_score),
        "went_to_ot": went_to_ot,
        "margin": round(margin),
        "total": round(total),
        "blowout_risk_factor": round(blowout_risk, 3),
        "pace_adjustment_factor": round(pace_adj, 4),
        "vegas_spread": round(home_score - away_score, 1),
        "game_total": round(total, 1),
        "environment_label": _classify_environment(margin, total, went_to_ot),
    }

def _classify_environment(margin, total, went_to_ot):
    if went_to_ot: return "? Overtime Thriller"
    if margin > 20: return "?? Blowout"
    if total > 235: return "?? Shootout"
    if total < 200: return "?? Defensive Grind"
    if margin < 5: return "?? Nail-Biter"
    return "?? Standard Game"

from engine.simulation import run_quantum_matrix_simulation

# Stat-type coefficient of variation (matches your engine's implicit CV values)
STAT_CV = {
    "points":    0.25,
    "rebounds":  0.30,
    "assists":   0.35,
    "steals":    0.50,
    "blocks":    0.55,
    "turnovers": 0.35,
    "threes":    0.40,
}

def simulate_player_stat(
    profile: dict,
    stat_type: str,
    env: dict,
    seed: int,
) -> int:
    """
    Simulate one stat for one player by calling YOUR existing QME.
    
    100 micro-sims ? sort ? pick percentile based on Clutch attribute.
    """
    stat_means = {
        "points":    profile["ppg"],
        "rebounds":  profile["rpg"],
        "assists":   profile["apg"],
        "steals":    profile["spg"],
        "blocks":    profile["bpg"],
        "turnovers": profile["tpg"],
        "threes":    profile["threes_pg"],
    }
    
    mean = stat_means[stat_type]
    if mean <= 0:
        return 0
    
    # Consistency attribute modifies std dev
    # High consistency (99) ? cv * 0.61, Low (30) ? cv * 1.16
    consistency = profile["attr_consistency"]
    cv_modifier = 1.0 + (50 - consistency) * 0.008
    cv_modifier = max(0.5, min(1.5, cv_modifier))
    
    std_dev = mean * STAT_CV[stat_type] * cv_modifier
    std_dev = max(0.5, std_dev)
    
    # Load game logs from player_game_logs table if available (for KDE/C11)
    recent_logs = _load_recent_stat_logs(profile["player_id"], stat_type)
    
    # ?? CALL YOUR EXISTING ENGINE ??
    result = run_quantum_matrix_simulation(
        projected_stat_average=mean,
        stat_standard_deviation=std_dev,
        prop_line=0.0,                          # Not evaluating a prop
        number_of_simulations=100,              # 100 micro-sims
        blowout_risk_factor=env["blowout_risk_factor"],      # FROM ENVIRONMENT
        pace_adjustment_factor=env["pace_adjustment_factor"], # FROM ENVIRONMENT
        matchup_adjustment_factor=1.0,          # NEUTRAL (pure simulation)
        home_away_adjustment=0.0,               # NEUTRAL
        rest_adjustment_factor=1.0,             # NEUTRAL
        stat_type=stat_type,
        projected_minutes=profile["minutes_pg"],
        minutes_std=5.0,                        # Your MINUTES_STD_DEFAULT
        recent_game_logs=recent_logs,           # KDE when 15+ logs available
        random_seed=seed,                       # DETERMINISTIC
        enable_fatigue_curve=True,              # Keep your Q1-Q4 decay
        vegas_spread=env["vegas_spread"],        # FROM ENVIRONMENT ? feeds 1A matrix
        game_total=env["game_total"],            # FROM ENVIRONMENT ? feeds 1A matrix
    )
    
    # ?? PERCENTILE PICK (attribute-driven) ??
    sorted_results = sorted(result["simulated_results"])
    
    import random as rng
    rng.seed(seed + 99999)
    
    clutch = profile["attr_clutch"]
    # High Clutch ? sample from upper-middle range (consistent upside)
    # Low Clutch ? wider range (unpredictable)
    low_pct = max(5, 50 - clutch // 2)    # 99?1, 50?25, 30?35
    high_pct = min(95, 50 + clutch // 2)   # 99?99, 50?75, 30?65
    
    # Hot/cold form shifts the range
    if profile.get("hot_cold_label") == "hot":
        low_pct = min(90, low_pct + 8)
        high_pct = min(98, high_pct + 5)
    elif profile.get("hot_cold_label") == "cold":
        low_pct = max(2, low_pct - 5)
        high_pct = max(40, high_pct - 8)
    
    chosen_pct = rng.randint(low_pct, high_pct)
    idx = int(chosen_pct / 100 * len(sorted_results))
    idx = max(0, min(len(sorted_results) - 1, idx))
    
    return max(0, round(sorted_results[idx]))
